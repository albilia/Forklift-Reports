async function runOCR(base64) { document.getElementById("loader").style.display = "block"; try { const response = await fetch(API_URL, { method: "POST", body: JSON.stringify({ action: "processImage", image: base64 }) }); const data = await response.json(); document.getElementById("loader").style.display = "none"; if (!data.success) { alert("שגיאה בפענוח התמונה"); return; } ocrResult = data.parsed; showOCR(data.parsed); } catch (err) { alert("שגיאה בחיבור לשרת"); document.getElementById("loader").style.display = "none"; } }
